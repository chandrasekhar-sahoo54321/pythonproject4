Create :-

===================

Example :-

=======================


import cx_Oracle

try:

con=cx_Oracle.connect('scott/tiger@localhost')

cursor=con.cursor()

cursor.execute("create table employees(eno number,ename varchar2(10),esal number(10,2),eaddr varchar2(10))")

print("Table created successfully")

except cx_Oracle.DatabaseError as e:

if con:

con.rollback()

print("There is a problem with sql",e)

finally:

if cursor:

cursor.close()

if con:

con.close()

=======================

Example :-

=======================


import cx_Oracle

try:

con=cx_Oracle.connect('scott/tiger@localhost')

cursor=con.cursor()

sql="insert into employees values(:eno,:ename,:esal,:eaddr)"

records=[(200,'Sunny',2000,'Mumbai'), 7) (300,'Chinny',3000,'Hyd'), 8) (400,'Bunny',4000,'Hyd')]

con.commit()

print("Records Inserted Successfully")

except cx_Oracle.DatabaseError as e:

if cursor:
